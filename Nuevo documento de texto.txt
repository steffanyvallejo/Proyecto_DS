CREATE TABLE t_pagoEfectivo (
    id_pagoC serial primary key NOT NULL,   
    subtotal float NOT NULL,
    impuesto float NOT NULL,
    cantidad float NOT NULL
);

CREATE TABLE t_pagoCredito (
    id_pagoC serial primary key NOT NULL,   
    subtotal float NOT NULL,
    impuesto float NOT NULL,
    nomTarjeta varchar(50),
    numTarjeta varchar(50) NOT NULL,
    fechaExp date NOT NULL
);

CREATE TABLE t_local (
    id_local serial primary key NOT NULL,
    direccion varchar(50) NOT NULL,
    telefono varchar(20) NOT NULL,
    internet boolean
);

CREATE TABLE t_cotizacion (
    id_cot serial primary key NOT NULL,
    cantidad int NOT NULL,
    valor float NOT NULL
);

CREATE TABLE t_venta (
    id_venta serial primary key NOT NULL,
    valor float NOT NULL,
    fecha date NOT NULL
);

CREATE TABLE t_cliente (
    id_cliente serial primary key NOT NULL,
    nombre varchar(50) NOT NULL,
    apellido varchar(50) NOT NULL,
    fecha_nacimiento date NOT NULL,
    telefono varchar(20) NOT NULL,
    direccion varchar(50) NOT NULL,
    correo varchar(50) NOT NULL
);

CREATE TABLE t_usuario (
    id_user serial primary key NOT NULL,
    nombre varchar(50) NOT NULL,
    apellido varchar(50) NOT NULL,
    fecha_nacimiento date NOT NULL,
    telefono varchar(20) NOT NULL,
    direccion varchar(50) NOT NULL,
    correo varchar(50) NOT NULL,
    usuario varchar(50) NOT NULL,
    pass varchar(50) NOT NULL,
    tipo varchar(10) NOT NULL,
    super boolean
);

CREATE TABLE t_articulo (
    id_art serial primary key NOT NULL,
    nombre varchar(50) NOT NULL,
    descripcion varchar(100) NOT NULL,
    categoria date NOT NULL,
    stock int NOT NULL,
    estado boolean NOT NULL,
    precio float NOT NULL,
    marca varchar(50) NOT NULL
);
